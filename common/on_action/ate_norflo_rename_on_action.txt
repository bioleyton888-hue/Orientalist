on_title_gain = {
	on_actions = {
		on_norflo_gain_rename
		on_norflo_gain_reset
	}
}

# Rename k_norflo to al-Norflo when a Neomoor or Shriner takes it
on_norflo_gain_rename = {
	trigger = {
		scope:title = { this = title:k_norflo }
		OR = {
			culture = culture:neomoor
			faith = faith:shriner
		}
	}
	effect = {
		title:k_norflo = { set_title_name = k_norflo_neomoor_shriner }
	}
}

# Reset k_norflo name when a non-Neomoor non-Shriner takes it
on_norflo_gain_reset = {
	trigger = {
		scope:title = { this = title:k_norflo }
		NOT = {
			OR = {
				culture = culture:neomoor
				faith = faith:shriner
			}
		}
	}
	effect = {
		title:k_norflo = { reset_title_name = yes }
	}
}
