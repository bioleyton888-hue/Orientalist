# On birth: grant Ya-Shamal Descendant and Mawlid al-Fez traits
on_birth_child = {
	events = {
		ate_shriner.0002
	}
	on_actions = {
		neomoor_mawlid_birth_check
	}
}

neomoor_mawlid_birth_check = {
	trigger = {
		OR = {
			scope:father ?= { has_title = title:d_shriner }
			scope:mother ?= { has_title = title:d_shriner }
		}
		NOT = { has_trait = mawlid_al_fez }
	}
	effect = {
		add_trait = mawlid_al_fez
	}
}

	on_actions = {

			}

				set_culture = culture:neomoor
			}
		}
	}
}

# On title gain: consolidated handler for culture assignment and title renames
# Merged from 6 separate named on_actions — engine stops at first else_if match
on_title_gain = {
	on_actions = {
		neomoor_shriner_title_gain_handler
		ate_portia_conquest_handler
	}
}

on_title_lost = {
	on_actions = {
		ate_portia_title_lost_handler
	}
}

on_death = {
	on_actions = {
		ate_portia_death_handler
	}
}

ate_portia_title_lost_handler = {
	trigger = {
		this = character:caribe_seacole_portia_jamaica2
		exists = global_var:ate_florida_confed_active
		# Cheap gate: the title just lost is a tracked Florida duchy or a county within one
		OR = {
			scope:title = { this = title:d_miami }
			scope:title = { this = title:d_suncoast }
			scope:title = { this = title:d_eyefour }
			scope:title = { this = title:d_first_coast }
			scope:title = {
				tier = tier_county
				OR = {
					duchy = title:d_miami
					duchy = title:d_suncoast
					duchy = title:d_eyefour
					duchy = title:d_first_coast
				}
			}
		}
		# A grant or internal reshuffle is NOT a real loss — if the new holder is one of
		# Portia's vassals the land stays in her realm and the arc stays alive
		NOT = {
			scope:new_holder ?= {
				target_is_liege_or_above = character:caribe_seacole_portia_jamaica2
			}
		}
		# After this loss her REALM (vassal-held counties included) has no county left in
		# the tracked duchies. this != scope:title excludes the title being lost — at
		# on_title_lost it is still counted (see vanilla title_on_actions.txt).
		NOT = {
			any_realm_county = {
				this != scope:title
				OR = {
					duchy = title:d_miami
					duchy = title:d_suncoast
					duchy = title:d_eyefour
					duchy = title:d_first_coast
				}
			}
		}
	}
	effect = {
		# Consume the flag NOW so the other title losses in this same tick don't also schedule it
		remove_global_variable = ate_florida_confed_active
		random_player = { limit = { ate_florida_confed_recipient_trigger = yes } trigger_event = { id = ate_florida_disband.0001 days = 1 } }
	}
}

# Portia conquest events — fire when Portia gains d_miami, d_suncoast, d_eyefour, or d_first_coast,
# OR when a county gain leaves her realm in complete control of one of those duchies without the
# duchy title itself (the AI may conquer county-by-county and never create/usurp the duchy).
# Per-duchy took-flags keep the county path and a later title creation from double-firing.
ate_portia_conquest_handler = {
	trigger = {
		this = character:caribe_seacole_portia_jamaica2
		OR = {
			scope:title = { this = title:d_miami }
			scope:title = { this = title:d_suncoast }
			scope:title = { this = title:d_eyefour }
			scope:title = { this = title:d_first_coast }
			scope:title = {
				tier = tier_county
				OR = {
					duchy = title:d_miami
					duchy = title:d_suncoast
					duchy = title:d_eyefour
					duchy = title:d_first_coast
				}
			}
		}
	}
	effect = {
		# Fire per-duchy flavor event on the player the day after Portia takes the duchy —
		# either the title itself, or full de jure control of it via a county gain
		if = {
			limit = {
				NOT = { has_character_flag = ate_portia_took_miami }
				OR = {
					scope:title = { this = title:d_miami }
					AND = {
						scope:title = { tier = tier_county duchy = title:d_miami }
						completely_controls = title:d_miami
					}
				}
			}
			add_character_flag = ate_portia_took_miami
			random_player = { limit = { ate_florida_confed_recipient_trigger = yes } trigger_event = { id = ate_portia.0001 days = 1 } }
		}
		else_if = {
			limit = {
				NOT = { has_character_flag = ate_portia_took_suncoast }
				OR = {
					scope:title = { this = title:d_suncoast }
					AND = {
						scope:title = { tier = tier_county duchy = title:d_suncoast }
						completely_controls = title:d_suncoast
					}
				}
			}
			add_character_flag = ate_portia_took_suncoast
			random_player = { limit = { ate_florida_confed_recipient_trigger = yes } trigger_event = { id = ate_portia.0002 days = 1 } }
		}
		else_if = {
			limit = {
				NOT = { has_character_flag = ate_portia_took_eyefour }
				OR = {
					scope:title = { this = title:d_eyefour }
					AND = {
						scope:title = { tier = tier_county duchy = title:d_eyefour }
						completely_controls = title:d_eyefour
					}
				}
			}
			add_character_flag = ate_portia_took_eyefour
			random_player = { limit = { ate_florida_confed_recipient_trigger = yes } trigger_event = { id = ate_portia.0003 days = 1 } }
		}
		else_if = {
			limit = {
				NOT = { has_character_flag = ate_portia_took_first_coast }
				OR = {
					scope:title = { this = title:d_first_coast }
					AND = {
						scope:title = { tier = tier_county duchy = title:d_first_coast }
						completely_controls = title:d_first_coast
					}
				}
			}
			add_character_flag = ate_portia_took_first_coast
			random_player = { limit = { ate_florida_confed_recipient_trigger = yes } trigger_event = { id = ate_portia.0004 days = 1 } }
		}
		# On the first duchy taken (by title or by full county control), schedule the
		# confederation event on the player after 30 days
		if = {
			limit = {
				NOT = { has_character_flag = ate_portia_confederation_scheduled }
				OR = {
					has_character_flag = ate_portia_took_miami
					has_character_flag = ate_portia_took_suncoast
					has_character_flag = ate_portia_took_eyefour
					has_character_flag = ate_portia_took_first_coast
				}
			}
			add_character_flag = ate_portia_confederation_scheduled
			random_player = { limit = { ate_florida_confed_recipient_trigger = yes } trigger_event = { id = ate_florida_confed.0001 days = 30 } }
		}
	}
}

# Ibrahim the Mad chain — fires for lunatic/possessed/eccentric/deviant/rakish/seducer Shriner emperors
# Trigger order optimized: cheap checks first, title/spouse iterators last
ate_ibrahim_obsession_pulse = {
	trigger = {
		faith = faith:shriner
		is_male = yes
		OR = {
			has_trait = lunatic_genetic
			has_trait = possessed_genetic
			has_trait = eccentric
			has_trait = deviant
			has_trait = rakish
			has_trait = seducer
		}
		NOT = { has_character_flag = ate_ibrahim_chain_started }
		NOT = { has_character_flag = ate_ibrahim_chain_done }
		NOT = { has_character_flag = ate_ibrahim_refused }
		any_held_title = { tier >= tier_empire }
		any_spouse = { count < 4 }
	}
	effect = {
		random = {
			chance = 10
			add_character_flag = ate_ibrahim_chain_started
			trigger_event = { id = ate_shriner.0005 }
		}
	}
}

# Post-conversion pulse — fires for 15 years after any Neomoor conversion decision, 3-year cooldown
ate_neomoor_post_conversion_pulse = {
	trigger = {
		culture = culture:neomoor
		has_character_flag = ate_neomoor_conversion_recent
		is_landed = yes
		is_adult = yes
		NOT = { has_character_flag = ate_post_conversion_cooldown }
	}
	effect = {
		random = {
			chance = 15
			add_character_flag = {
				flag = ate_post_conversion_cooldown
				years = 3
			}
			random_list = {
				30 = {
					modifier = {
						add = -1000000
						NOT = {
							OR = {
								AND = {
									exists = title:e_hcc
									NOT = { has_title = title:e_hcc }
								}
								any_neighboring_realm_owner = {
									NOT = { culture = culture:neomoor }
									NOT = { faith = faith:shriner }
									is_landed = yes
								}
							}
						}
					}
					trigger_event = { id = ate_neomoor.0001 }
				}
				35 = {
					modifier = {
						add = -1000000
						NOT = {
							any_vassal = {
								OR = {
									culture = culture:sunshiner
									culture = culture:sofloano
								}
							}
						}
					}
					trigger_event = { id = ate_neomoor.0002 }
				}
				25 = {
					trigger_event = { id = ate_neomoor.0003 }
				}
			}
		}
	}
}
