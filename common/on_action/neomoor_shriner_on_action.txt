# On birth: grant Ya-Shamal Descendant trait to children of Ya-Shamal holders
on_birth_child = {
	events = {
		ate_shriner.0002
	}
}

# On game start: set the initial Imperial Divan head to NeoMoor culture
on_game_start = {
	on_actions = {
		neomoor_shriner_set_initial_culture
	}
}

neomoor_shriner_set_initial_culture = {
	effect = {
		if = {
			limit = {
				exists = character:religious_shriner_hof_1
			}
			character:religious_shriner_hof_1 = {
				set_culture = culture:neomoor
			}
		}
	}
}

# Yearly pulse: trigger Badroulbadour chain for Nelson Shriners
# Also triggers Ibrahim the Mad chain for eccentric/deviant Shriner emperors
yearly_playable_pulse = {
	on_actions = {
		nelson_parade_check
		badroulbadour_chain_check
		ate_ibrahim_obsession_pulse
	}
}

nelson_parade_check = {
	trigger = {
		dynasty = dynasty:dynn_caribe_nelson
		faith = faith:shriner
		is_adult = yes
		is_landed = yes
		is_male = yes
		NOT = { has_character_flag = nelson_parade_started }
		NOT = { exists = global_var:nelson_parade_performed }
	}
	effect = {
		random = {
			chance = 50
			trigger_event = { id = ate_nelson.0009 }
		}
	}
}

badroulbadour_chain_check = {
	trigger = {
		is_player_character = yes
		faith = faith:shriner
		is_adult = yes
		is_landed = yes
		is_male = yes
		NOT = { has_character_flag = badroulbadour_started }
		NOT = { has_character_flag = badroulbadour_done }
		NOT = { exists = global_var:ya_shamal_performed }
		OR = {
			AND = {
				dynasty = dynasty:dynn_caribe_nelson
				has_character_flag = nelson_parade_done
			}
			AND = {
				NOT = { dynasty = dynasty:dynn_caribe_nelson }
				any_held_title = { tier >= tier_kingdom }
			}
		}
	}
	effect = {
		random = {
			chance = 20
			trigger_event = { id = ate_badroulbadour.0001 }
		}
	}
}

# On title gain: consolidated handler for culture assignment and title renames
# Merged from 6 separate named on_actions — engine stops at first else_if match
on_title_gain = {
	on_actions = {
		neomoor_shriner_title_gain_handler
	}
}

neomoor_shriner_title_gain_handler = {
	effect = {
		if = {
			limit = { scope:title = { this = title:d_shriner } }
			set_culture = culture:neomoor
		}
		else_if = {
			limit = { scope:title = { this = title:k_florida } }
			if = {
				limit = {
					OR = {
						culture = culture:neomoor
						faith = faith:shriner
					}
				}
				title:k_florida = { set_title_name = k_florida_agrabah }
			}
			else = {
				title:k_florida = { set_title_name = k_florida }
			}
		}
		else_if = {
			limit = { scope:title = { this = title:k_louisiana } }
			if = {
				limit = {
					OR = {
						culture = culture:neomoor
						faith = faith:shriner
					}
				}
				title:k_louisiana = { set_title_name = k_louisiana_neomoor }
			}
			else = {
				title:k_louisiana = { set_title_name = k_louisiana }
			}
		}
		else_if = {
			limit = { scope:title = { this = title:k_soflo } }
			if = {
				limit = {
					OR = {
						culture = culture:neomoor
						faith = faith:shriner
					}
				}
				title:k_soflo = { set_title_name = k_soflo_neomoor_shriner }
			}
			else = {
				title:k_soflo = { set_title_name = k_soflo }
			}
		}
		else_if = {
			limit = { scope:title = { this = title:k_norflo } }
			if = {
				limit = {
					OR = {
						culture = culture:neomoor
						faith = faith:shriner
					}
				}
				title:k_norflo = { set_title_name = k_norflo_neomoor_shriner }
			}
			else = {
				title:k_norflo = { set_title_name = k_norflo }
			}
		}
		else_if = {
			limit = { scope:title = { this = title:e_gulf_empire } }
			if = {
				limit = {
					OR = {
						culture = culture:neomoor
						faith = faith:shriner
					}
				}
				title:e_gulf_empire = { set_title_name = e_gulf_empire_neomoor }
			}
			else = {
				title:e_gulf_empire = { set_title_name = e_gulf_empire }
			}
		}
	}
}

# Ibrahim the Mad chain — fires for lunatic/possessed/eccentric/deviant/rakish/seducer Shriner emperors
# Trigger order optimized: cheap checks first, title/spouse iterators last
ate_ibrahim_obsession_pulse = {
	trigger = {
		is_player_character = yes
		faith = faith:shriner
		is_male = yes
		OR = {
			has_trait = lunatic
			has_trait = possessed
			has_trait = eccentric
			has_trait = deviant
			has_trait = rakish
			has_trait = seducer
		}
		NOT = { has_character_flag = ate_ibrahim_chain_started }
		NOT = { has_character_flag = ate_ibrahim_chain_done }
		NOT = { has_character_flag = ate_ibrahim_refused }
		any_held_title = { tier >= tier_empire }
		any_spouse = { count < 4 }
	}
	effect = {
		random = {
			chance = 10
			add_character_flag = ate_ibrahim_chain_started
			trigger_event = { id = ate_shriner.0005 }
		}
	}
}
