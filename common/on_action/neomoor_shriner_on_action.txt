# On birth: grant Ya-Shamal Descendant trait to children of Ya-Shamal holders
on_birth_child = {
	events = {
		ate_shriner.0002
	}
}

# On game start: set the initial Imperial Divan head to NeoMoor culture
on_game_start = {
	on_actions = {
		neomoor_shriner_set_initial_culture
	}
}

neomoor_shriner_set_initial_culture = {
	effect = {
		if = {
			limit = {
				exists = character:religious_shriner_hof_1
			}
			character:religious_shriner_hof_1 = {
				set_culture = culture:neomoor
			}
		}
	}
}

# Yearly pulse: trigger Badroulbadour chain for Nelson Shriners
# Also triggers Ibrahim the Mad chain for eccentric/deviant Shriner emperors
yearly_playable_pulse = {
	on_actions = {
		nelson_parade_check
		badroulbadour_chain_check
		ate_ibrahim_obsession_pulse
	}
}

nelson_parade_check = {
	trigger = {
		dynasty = dynasty:dynn_caribe_nelson
		faith = faith:shriner
		is_adult = yes
		is_landed = yes
		is_male = yes
		NOT = { has_character_flag = nelson_parade_started }
		NOT = { exists = global_var:nelson_parade_performed }
	}
	effect = {
		random = {
			chance = 50
			trigger_event = { id = ate_nelson.0009 }
		}
	}
}

badroulbadour_chain_check = {
	trigger = {
		is_player_character = yes
		faith = faith:shriner
		is_adult = yes
		is_landed = yes
		is_male = yes
		NOT = { has_character_flag = badroulbadour_started }
		NOT = { has_character_flag = badroulbadour_done }
		NOT = { exists = global_var:ya_shamal_performed }
		OR = {
			AND = {
				dynasty = dynasty:dynn_caribe_nelson
				has_character_flag = nelson_parade_done
			}
			AND = {
				NOT = { dynasty = dynasty:dynn_caribe_nelson }
				any_held_title = { tier >= tier_kingdom }
			}
		}
	}
	effect = {
		random = {
			chance = 20
			trigger_event = { id = ate_badroulbadour.0001 }
		}
	}
}

# On title gain: set NeoMoor culture on anyone who gains d_shriner
# Also rename k_florida to Agrabah if holder is Shriner or NeoMoor
on_title_gain = {
	on_actions = {
		neomoor_shriner_culture_on_title_gain
		neomoor_shriner_florida_rename_on_title_gain
		neomoor_louisiana_rename_on_title_gain
		neomoor_shriner_soflo_rename_on_title_gain
		neomoor_shriner_norflo_rename_on_title_gain
		neomoor_gulf_rename_on_title_gain
	}
}

neomoor_shriner_culture_on_title_gain = {
	trigger = {
		scope:title = { this = title:d_shriner }
	}
	effect = {
		set_culture = culture:neomoor
	}
}

neomoor_louisiana_rename_on_title_gain = {
	trigger = {
		scope:title = { this = title:k_louisiana }
	}
	effect = {
		if = {
			limit = {
				OR = {
					culture = culture:neomoor
					faith = faith:shriner
				}
			}
			title:k_louisiana = { set_title_name = k_louisiana_neomoor }
		}
		else = {
			title:k_louisiana = { set_title_name = k_louisiana }
		}
	}
}

# Ibrahim the Mad chain — fires for lunatic/possessed/eccentric/deviant/rakish/seducer Shriner emperors
ate_ibrahim_obsession_pulse = {
	trigger = {
		is_player_character = yes
		faith = faith:shriner
		is_male = yes
		any_held_title = { tier >= tier_empire }
		OR = {
			has_trait = lunatic
			has_trait = possessed
			has_trait = eccentric
			has_trait = deviant
			has_trait = rakish
			has_trait = seducer
		}
		any_spouse = { count < 4 }
		NOT = { has_character_flag = ate_ibrahim_chain_started }
		NOT = { has_character_flag = ate_ibrahim_chain_done }
		NOT = { has_character_flag = ate_ibrahim_refused }
	}
	effect = {
		random = {
			chance = 10
			add_character_flag = ate_ibrahim_chain_started
			trigger_event = { id = ate_shriner.0005 }
		}
	}
}
