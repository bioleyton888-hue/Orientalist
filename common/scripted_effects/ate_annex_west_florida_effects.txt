# ATE Neomoor — Annex West Florida effect, called by our decision
# (common/decisions/ate_annex_west_florida.txt). Mirror of Assert Gulf Independence, but for
# the three powers that can swallow k_west_florida back up instead of letting the Gulf keep it.
# Caribbean/HCC: take the kingdom wholesale (transfer holder + de jure liege), same pattern as
# Assert Gulf Independence. Mississippi: TRUE RECLAMATION — destroy the kingdom shell and restore
# every duchy/county to its pre-Proclaim de jure parent (reverses
# ate_proclaim_west_florida_decision_effects exactly). Root = the annexer (scope:annex_west_florida_taker).
ate_annex_west_florida_caribbean_effects = {
	scope:annex_west_florida_taker = {
		create_title_and_vassal_change = {
			type = conquest
			save_scope_as = title_change
			add_claim_on_loss = no
		}
		title:k_west_florida = {
			change_title_holder = {
				holder = scope:annex_west_florida_taker
				change = scope:title_change
			}
		}
		resolve_title_and_vassal_change = scope:title_change

		title:k_west_florida = { set_de_jure_liege_title = title:e_caribbean }
	}
}

# The Gulf Empire re-taking its own cradle: the empire was BORN from k_west_florida (Assert Gulf
# Independence requires holding it), so this branch only fires after having it, losing it to a
# vassal, and taking it back — the Stalingrad railway station of the Gulf coast. The de jure
# re-affirm is idempotent when nothing moved it, and heals it if some other decision did.
ate_annex_west_florida_gulf_effects = {
	scope:annex_west_florida_taker = {
		create_title_and_vassal_change = {
			type = conquest
			save_scope_as = title_change
			add_claim_on_loss = no
		}
		title:k_west_florida = {
			change_title_holder = {
				holder = scope:annex_west_florida_taker
				change = scope:title_change
			}
		}
		resolve_title_and_vassal_change = scope:title_change

		title:k_west_florida = { set_de_jure_liege_title = title:e_gulf_empire }
	}
}

# Mississippi does not take the land for itself — it dissolves the kingdom shell and hands every
# piece back to the de jure parent it was carved from. No holder change: whoever currently holds
# each duchy/county keeps it, only the de jure map and the empty k_west_florida title are touched.
ate_annex_west_florida_mississippi_effects = {
	title:d_west_florida = { set_de_jure_liege_title = title:k_louisiana }
	title:d_homochitto = { set_de_jure_liege_title = title:k_magnolia }
	title:d_pine_belt = { set_de_jure_liege_title = title:k_magnolia }
	title:d_conecuh = { set_de_jure_liege_title = title:k_alabama }
	title:d_wiregrass = { set_de_jure_liege_title = title:k_alabama }
	title:d_florabama = { set_de_jure_liege_title = title:k_norflo }
	title:c_mobile.de_jure_liege = { set_de_jure_liege_title = title:k_alabama }
	title:c_demopolis = { set_de_jure_liege_title = title:d_canebreak }

	# Now landless (its de jure duchies are gone) — remove the empty shell.
	destroy_title = title:k_west_florida
}
