# Nelson Neomoor — Culture transformation upon controlling SoFlo
ate_nelson_soflo_neomoor = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_dynasty_houses.dds"
	}
	decision_group_type = major

	ai_check_interval = 60

	desc = ate_nelson_soflo_neomoor_desc
	selection_tooltip = ate_nelson_soflo_neomoor_tooltip

	is_shown = {
		dynasty = dynasty:dynn_caribe_nelson
		NOT = { has_character_flag = nelson_neomoor_done }
	}

	is_valid = {
		has_title = title:k_soflo
		years_as_ruler >= 5
		# Totality of SoFlo — all de jure counties of k_soflo under root
		NOT = {
			title:k_soflo = {
				any_de_jure_county = {
					NOT = { holder.top_liege = root }
				}
			}
		}
	}

	is_valid_showing_failures_only = {
		is_available_adult = yes
	}

	effect = {
		trigger_event = { id = ate_nelson.0001 }
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 10
	}
}
