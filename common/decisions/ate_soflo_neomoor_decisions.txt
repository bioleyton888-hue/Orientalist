# Nelson Neomoor — Culture transformation upon controlling SoFlo
ate_nelson_soflo_neomoor = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_culture.dds"
	}
	decision_group_type = major

	ai_check_interval = 60

	desc = ate_nelson_soflo_neomoor_desc
	selection_tooltip = ate_nelson_soflo_neomoor_tooltip

	is_shown = {
		dynasty = dynasty:dynn_caribe_nelson
		NOT = { exists = global_var:ate_nelson_neomoor_done }
	}

	is_valid = {
		OR = {
			AND = {
				has_title = title:k_soflo
				NOT = {
					title:k_soflo = {
						any_de_jure_county = {
							NOT = { holder.top_liege = root }
						}
					}
				}
			}
			AND = {
				has_title = title:k_florida
				NOT = {
					title:k_florida = {
						any_de_jure_county = {
							NOT = { holder.top_liege = root }
						}
					}
				}
			}
		}
		custom_tooltip = {
			text = ate_soflo_years_as_ruler_tt
			years_as_ruler >= 5
		}
	}

	is_valid_showing_failures_only = {
		is_available_adult = yes
		custom_tooltip = {
			text = ate_soflo_years_as_ruler_tt
			years_as_ruler >= 5
		}
	}

	effect = {
		set_global_variable = ate_nelson_neomoor_done
		add_character_flag = { flag = ate_neomoor_conversion_recent years = 15 }
		add_character_modifier = { modifier = ate_nelson_neomoor_conversion years = -1 }
		set_culture = culture:neomoor
		culture:neomoor = {
			if = {
				limit = { has_cultural_tradition = tradition_diasporic }
				remove_culture_tradition = tradition_diasporic
			}
		}
		every_child = {
			limit = { NOT = { culture = culture:neomoor } }
			set_culture = culture:neomoor
		}
		every_vassal_or_below = {
			limit = { faith = faith:shriner }
			set_culture = culture:neomoor
			every_child = {
				limit = { NOT = { culture = culture:neomoor } }
				set_culture = culture:neomoor
			}
		}
		title:c_tampa = { set_county_culture = culture:neomoor }
		title:c_st_petersburg = { set_county_culture = culture:neomoor }
		trigger_event = { id = ate_nelson.0001 }
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 10
	}
}
