####################################################################################
##  ATE NEOMOOR  —  Assert Gulf Independence  (re-adds the decision 0.21.2 removed)
##  Creates the Empire of the Gulf (e_gulf_empire / "al-Hijrah") and folds West Florida
##  + Florida (+ Louisiana stretch) under it as de jure. Meant to FOLLOW Proclaim West
##  Florida: it requires you already hold a real k_west_florida (d_west_florida de jure
##  under it) — exactly what our Proclaim West Florida builds.
##  PORTED FROM After the End Dev 1.17 (commit 26e37a18). Adapted for 0.21.2: removed all
##  k_florida_waltney references (that title was deleted by the dynamic-naming rework).
##  NOTE: e_gulf_empire title already exists in base — we reference it, never re-declare it.
####################################################################################
assert_gulf_independence_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
	}
	decision_group_type = major
	ai_check_interval = 12
	desc = assert_gulf_independence_decision_desc
	selection_tooltip = assert_gulf_independence_decision_tooltip
	is_shown = {
		NOT = {  #Can only do it once.
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:assert_gulf_independence_decision
			}
		}
		OR = {
			culture = culture:gulfard
			culture = {
				any_parent_culture_or_above = {
					this = culture:gulfard
				}
			}
			culture = culture:sunshiner
			culture = {
				any_parent_culture_or_above = {
					this = culture:sunshiner
				}
			}
			culture = culture:sofloano
			culture = {
				any_parent_culture_or_above = {
					this = culture:sofloano
				}
			}
		}
		# Dropped a redundant highest_held_title_tier >= tier_kingdom here: every title in the
		# OR below is a kingdom, so holding one already guarantees kingdom tier.
		OR = {
			has_title = title:k_norflo
			has_title = title:k_soflo
			has_title = title:k_florida
		}
	}
	is_valid = {
		is_independent_ruler = yes
		# Dropped a redundant highest_held_title_tier >= tier_kingdom here: completely_controls =
		# title:k_florida below already requires holding the Florida kingdom (= kingdom tier).
		OR = {
			custom_tooltip = {
				text = assert_gulf_independence_primary_title_florida
				primary_title = title:k_florida
			}
			custom_tooltip = {
				text = assert_gulf_independence_primary_title_west_florida
				primary_title = title:k_west_florida
			}
		}
		AND = {
			completely_controls = title:k_west_florida
			title:d_west_florida.de_jure_liege = title:k_west_florida
		}
		AND = {
			completely_controls = title:k_florida
			title:d_miami.de_jure_liege = title:k_florida
		}
		custom_tooltip = {
			text = assert_gulf_independence_decision_stretch_goals
		}
	}

	effect = {
		save_scope_as = assert_gulf_independence_former
		hidden_effect = {
			add_to_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:assert_gulf_independence_decision
			}
			set_global_variable = {
				name = assert_gulf_independence_decision
				value = root
			}
		}
		if = {
			limit = { has_dlc_feature = legends }
			create_legend_seed = {
				type = legitimizing
				quality = illustrious
				chronicle = new_title
				properties = {
					title = title:e_gulf_empire
					founder = root
				}
			}
		}
		ate_assert_gulf_independence_decision_effects = yes
	}

	is_valid_showing_failures_only = {
		is_capable_adult = yes
		is_imprisoned = no
		is_independent_ruler = yes
	}

	cost = {
		prestige = { value = 750 }
		gold = { value = 750 }
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}
