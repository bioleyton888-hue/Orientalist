namespace = ate_floridian

# Floridian Pet alligator chain
#	ate_floridian.0001 - A Swamp Hatchling (discovery)
#	ate_floridian.0002 - Name the alligator
#	ate_floridian.0003 - The alligator settles in (modifier granted)
#
# Council follow-up chain
#	ate_floridian.0004 - Bring the gator to the council meeting?
#	ate_floridian.0005 - (A 75%) Gator intimidates the least-loyal councillor
#	ate_floridian.0006 - (A 25%) Gator bites the least-loyal councillor
#	ate_floridian.0007 - (B 50%) Gator plays with a courtier's children
#	ate_floridian.0008 - (B 50%) Gator eats a servant

# A Swamp Hatchling
ate_floridian.0001 = {
	type = character_event
	title = ate_floridian.0001.t
	desc = ate_floridian.0001.desc
	theme = pet
	left_portrait = {
		character = root
		animation = shock
	}
	override_background = { reference = forest }

	trigger = {
		is_player_character = yes
		culture = { has_cultural_heritage = heritage_floridian }
		NOT = { has_character_modifier = ate_pet_alligator_modifier }
		NOT = { has_character_flag = ate_gator_declined }
		is_available_healthy_adult = yes
	}

	immediate = {
		play_music_cue = "mx_cue_low_key_positive"
	}

	option = {
		name = ate_floridian.0001.a
		trigger_event = { id = ate_floridian.0002 }
		ai_chance = { base = 100 }
	}

	option = {
		name = ate_floridian.0001.b
		add_character_flag = {
			flag = ate_gator_declined
			years = 10
		}
		ai_chance = { base = 0 }
	}
}

# Name your alligator
ate_floridian.0002 = {
	type = character_event
	title = ate_floridian.0002.t
	desc = ate_floridian.0002.desc
	theme = pet
	left_portrait = {
		character = root
		animation = happiness
	}
	override_background = { reference = forest }

	immediate = {
		add_character_flag = ate_gator_default_wally
		add_character_flag = ate_gator_default_ali
		add_character_flag = ate_gator_default_waylon
		add_character_flag = ate_gator_default_snapper

		if = {
			limit = { is_ai = yes }
			random_list = {
				1 = { set_variable = { name = ate_gator_name  value = flag:ate_gator_name_wally } }
				1 = { set_variable = { name = ate_gator_name  value = flag:ate_gator_name_ali } }
				1 = { set_variable = { name = ate_gator_name  value = flag:ate_gator_name_waylon } }
				1 = { set_variable = { name = ate_gator_name  value = flag:ate_gator_name_snapper } }
			}
		}
	}

	widget = {
		gui = "event_window_widget_enter_text"
		container = "dynamic_birth_name"
		controller = {
			type = text
			data = {
				key = ate_gator_name
				default = {
					localization_key = ate_gator_name_wally
					trigger = { has_character_flag = ate_gator_default_wally }
				}
				default = {
					localization_key = ate_gator_name_ali
					trigger = { has_character_flag = ate_gator_default_ali }
				}
				default = {
					localization_key = ate_gator_name_waylon
					trigger = { has_character_flag = ate_gator_default_waylon }
				}
				default = {
					localization_key = ate_gator_name_snapper
					trigger = { has_character_flag = ate_gator_default_snapper }
				}
			}
		}
		setup_scope = {
			root = { save_scope_as = text_target }
		}
	}

	option = {
		name = ate_floridian.0002.a
		trigger_event = { id = ate_floridian.0003 }
	}

	after = {
		remove_character_flag = ate_gator_default_wally
		remove_character_flag = ate_gator_default_ali
		remove_character_flag = ate_gator_default_waylon
		remove_character_flag = ate_gator_default_snapper
		if = {
			limit = { NOT = { has_variable = ate_gator_name } }
			set_variable = {
				name = ate_gator_name
				value = flag:ate_gator_name_wally
			}
		}
	}
}

# Bring the gator to the council meeting?
ate_floridian.0004 = {
	type = character_event
	title = ate_floridian.0004.t
	desc = ate_floridian.0004.desc
	theme = pet
	left_portrait = {
		character = root
		animation = thinking
	}
	override_background = { reference = council_chamber }

	trigger = {
		is_player_character = yes
		has_character_modifier = ate_pet_alligator_modifier
		is_landed = yes
		NOT = { has_character_flag = ate_gator_council_cooldown }
	}

	immediate = {
		add_character_flag = {
			flag = ate_gator_council_cooldown
			years = 10
		}
	}

	option = {
		name = ate_floridian.0004.a
		every_councillor = {
			custom = every_councillor
			add_opinion = {
				target = root
				modifier = respect_opinion
				opinion = -5
			}
		}
		random_list = {
			75 = {
				desc = ate_floridian.0004.a_success
				show_chance = no
				custom_tooltip = ate_floridian.0004.a_success.tt
				trigger_event = {
					id = ate_floridian.0005
					days = { 7 10 }
				}
			}
			25 = {
				desc = ate_floridian.0004.a_failure
				show_chance = no
				custom_tooltip = ate_floridian.0004.a_failure.tt
				trigger_event = {
					id = ate_floridian.0006
					days = { 7 10 }
				}
			}
		}
	}

	option = {
		name = ate_floridian.0004.b
		random_list = {
			50 = {
				desc = ate_floridian.0004.b_play
				show_chance = no
				custom_tooltip = ate_floridian.0004.b_play.tt
				trigger = {
					any_courtier = {
						any_child = { is_courtier_of = root }
					}
				}
				trigger_event = {
					id = ate_floridian.0007
					days = { 7 10 }
				}
			}
			50 = {
				desc = ate_floridian.0004.b_eat
				show_chance = no
				custom_tooltip = ate_floridian.0004.b_eat.tt
				trigger_event = {
					id = ate_floridian.0008
					days = { 7 10 }
				}
			}
		}
	}
}

# A 75% — Gator intimidates the least-loyal councillor
ate_floridian.0005 = {
	type = character_event
	title = ate_floridian.0004.t
	desc = ate_floridian.0005.desc
	theme = pet
	override_background = { reference = council_chamber }
	left_portrait = {
		character = root
		animation = happiness
	}
	right_portrait = {
		character = scope:councillor
		animation = fear
	}

	immediate = {
		random_councillor = {
			weight = {
				base = 1
				opinion_modifier = {
					opinion_target = root
					multiplier = -5
				}
			}
			save_scope_as = councillor
		}
	}

	option = {
		name = ate_floridian.0005.a
		add_prestige = minor_prestige_gain
	}
}

# B 50% — Gator plays with a courtier's children
ate_floridian.0007 = {
	type = character_event
	title = ate_floridian.0004.t
	desc = ate_floridian.0007.desc
	theme = pet
	override_background = { reference = courtyard }
	left_portrait = {
		character = scope:lucky_courtier
		animation = happiness
	}

	trigger = {
		any_courtier = {
			any_child = { is_courtier_of = root }
		}
	}

	immediate = {
		random_courtier = {
			limit = {
				any_child = { is_courtier_of = root }
			}
			save_scope_as = lucky_courtier
		}
	}

	option = {
		name = ate_floridian.0007.a
		scope:lucky_courtier = {
			add_opinion = {
				target = root
				modifier = friendliness_opinion
				opinion = 15
			}
			every_child = {
				limit = { is_courtier_of = root }
				add_opinion = {
					target = root
					modifier = friendliness_opinion
					opinion = 15
				}
			}
		}
	}
}
