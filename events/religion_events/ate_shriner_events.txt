# Ya-Shamal Descendant — hidden birth event
ate_shriner.0002 = {
	type = character_event
	hidden = yes

	trigger = {
		OR = {
			AND = {
				exists = scope:mother
				scope:mother = {
					OR = {
						has_trait = ya_shamal
						has_trait = ya_shamal_descendant
					}
				}
			}
			AND = {
				exists = scope:father
				scope:father = {
					OR = {
						has_trait = ya_shamal
						has_trait = ya_shamal_descendant
					}
				}
			}
		}
		NOR = {
			has_trait = ya_shamal
			has_trait = ya_shamal_descendant
		}
	}

	immediate = {
		add_trait = ya_shamal_descendant
	}
}
