namespace = ate_shriner_pilgrimage

# ——— Shriner Pilgrimage to Tampa ———
# Fires via shriner_tampa_arrival_events when a Shriner player arrives at the tampa_shriner holy site.
# Chain: 0001 (arrival + setup) → 0002 (dispatcher, hidden) → 4 of 6 pool events → 0100 (Imperial Potentate's temple) → 0101 (return)

# ate_shriner_pilgrimage.0001 — The Grand Bazaar of Tampa (arrival)
ate_shriner_pilgrimage.0001 = {
	type = activity_event
	title = ate_shriner_pilgrimage.0001.t
	desc = ate_shriner_pilgrimage.0001.desc
	theme = faith

	trigger = {
		is_player_character = yes
	}

	left_portrait = {
		character = root
		animation = personality_zealous
	}

	immediate = {
		add_character_flag = in_pilgrimage_event_chain
		# Grant pilgrim trait if not already one — our chain replaces the vanilla arrival
		if = {
			limit = { NOT = { has_trait = pilgrim } }
			add_trait = pilgrim
		}
		# Set all 6 pool flags
		add_character_flag = shriner_pilg_a
		add_character_flag = shriner_pilg_b
		add_character_flag = shriner_pilg_c
		add_character_flag = shriner_pilg_d
		add_character_flag = shriner_pilg_e
		add_character_flag = shriner_pilg_f
		# Remove 2 randomly — first exclusion (uniform from 6)
		random_list = {
			1 = { remove_character_flag = shriner_pilg_a }
			1 = { remove_character_flag = shriner_pilg_b }
			1 = { remove_character_flag = shriner_pilg_c }
			1 = { remove_character_flag = shriner_pilg_d }
			1 = { remove_character_flag = shriner_pilg_e }
			1 = { remove_character_flag = shriner_pilg_f }
		}
		# Second exclusion (uniform from remaining 5)
		random_list = {
			1 = {
				modifier = { add = -1000000 NOT = { has_character_flag = shriner_pilg_a } }
				remove_character_flag = shriner_pilg_a
			}
			1 = {
				modifier = { add = -1000000 NOT = { has_character_flag = shriner_pilg_b } }
				remove_character_flag = shriner_pilg_b
			}
			1 = {
				modifier = { add = -1000000 NOT = { has_character_flag = shriner_pilg_c } }
				remove_character_flag = shriner_pilg_c
			}
			1 = {
				modifier = { add = -1000000 NOT = { has_character_flag = shriner_pilg_d } }
				remove_character_flag = shriner_pilg_d
			}
			1 = {
				modifier = { add = -1000000 NOT = { has_character_flag = shriner_pilg_e } }
				remove_character_flag = shriner_pilg_e
			}
			1 = {
				modifier = { add = -1000000 NOT = { has_character_flag = shriner_pilg_f } }
				remove_character_flag = shriner_pilg_f
			}
		}
	}

	option = {
		name = ate_shriner_pilgrimage.0001.a
		trigger_event = { id = ate_shriner_pilgrimage.0002 days = 1 }
	}
}

# ate_shriner_pilgrimage.0002 — Dispatcher (hidden)
# Fires the next available pool event, or proceeds to the temple when pool is exhausted.
ate_shriner_pilgrimage.0002 = {
	type = activity_event
	hidden = yes

	immediate = {
		if = {
			limit = { has_character_flag = shriner_pilg_a }
			remove_character_flag = shriner_pilg_a
			trigger_event = { id = ate_shriner_pilgrimage.0010 days = 2 }
		}
		else_if = {
			limit = { has_character_flag = shriner_pilg_b }
			remove_character_flag = shriner_pilg_b
			trigger_event = { id = ate_shriner_pilgrimage.0020 days = 2 }
		}
		else_if = {
			limit = { has_character_flag = shriner_pilg_c }
			remove_character_flag = shriner_pilg_c
			trigger_event = { id = ate_shriner_pilgrimage.0030 days = 2 }
		}
		else_if = {
			limit = { has_character_flag = shriner_pilg_d }
			remove_character_flag = shriner_pilg_d
			trigger_event = { id = ate_shriner_pilgrimage.0040 days = 2 }
		}
		else_if = {
			limit = { has_character_flag = shriner_pilg_e }
			remove_character_flag = shriner_pilg_e
			trigger_event = { id = ate_shriner_pilgrimage.0050 days = 2 }
		}
		else_if = {
			limit = { has_character_flag = shriner_pilg_f }
			remove_character_flag = shriner_pilg_f
			trigger_event = { id = ate_shriner_pilgrimage.0060 days = 2 }
		}
		else = {
			trigger_event = { id = ate_shriner_pilgrimage.0100 days = 3 }
		}
	}
}
