# ate_badroulbadour.0001 — A Captivating Dance (first encounter)
ate_badroulbadour.0001 = {
	type = character_event
	title = ate_badroulbadour.0001.t
	desc = ate_badroulbadour.0001.desc
	theme = party

	immediate = {
		create_character = {
			name = "Badroulbadour"
			gender = female
			faith = faith:shriner
			culture = culture:neomoor
			age = 22
			trait = beauty_good_3
			trait = intellect_good_2
			trait = gregarious
			employer = root
			save_scope_as = badroulbadour_scope
		}
		set_variable = { name = badroulbadour value = scope:badroulbadour_scope }
		add_character_flag = badroulbadour_started
		scope:badroulbadour_scope = { set_house = house:house_ElShabaz }
	}

	left_portrait = {
		character = root
		animation = idle
	}
	right_portrait = {
		character = scope:badroulbadour_scope
		animation = personality_content
	}

	# Speak with her
	option = {
		name = ate_badroulbadour.0001.a
		trigger_event = { id = ate_badroulbadour.0002 days = 1 }
	}

	# Walk away
	option = {
		name = ate_badroulbadour.0001.b
		scope:badroulbadour_scope = { move_to_pool = yes }
		remove_variable = badroulbadour
		remove_character_flag = badroulbadour_started
		add_character_flag = badroulbadour_done
	}
}

# ate_badroulbadour.0005 — To Break A Curse
ate_badroulbadour.0005 = {
	type = character_event
	title = ate_badroulbadour.0005.t
	desc = ate_badroulbadour.0005.desc
	theme = love

	trigger = {
		exists = var:badroulbadour
		var:badroulbadour = { is_alive = yes }
	}

	immediate = {
		var:badroulbadour = { save_scope_as = badroulbadour_scope }
	}

	left_portrait = {
		character = root
		animation = personality_content
	}
	right_portrait = {
		character = scope:badroulbadour_scope
		animation = happiness
	}

	option = {
		name = ate_badroulbadour.0005.a
		if = {
			limit = {
				OR = {
					any_spouse = { count >= 4 }
					AND = {
						any_spouse = { count >= 3 }
						is_betrothed = yes
					}
				}
			}
			trigger_event = { id = ate_badroulbadour.0005b days = 3 }
		}
		else = {
			if = { limit = { is_betrothed = yes } break_betrothal = betrothed }
			scope:badroulbadour_scope = { marry = root }
			trigger_event = { id = ate_badroulbadour.0006 days = 30 }
		}
	}

	# Gay/Asexual — cannot bring himself to marry her
	option = {
		name = ate_badroulbadour.0005.b
		trigger = { OR = { has_sexuality = homosexual has_sexuality = asexual } }
		add_prestige = -20
		add_piety = -20
		scope:badroulbadour_scope = { move_to_pool = yes }
		remove_variable = badroulbadour
		add_character_flag = badroulbadour_done
	}

	# Cannibal — consume the curse
	option = {
		name = ate_badroulbadour.0005.c
		trigger = { has_trait = cannibal }
		trigger_event = { id = ate_badroulbadour.0007 days = 1 }
	}
}

# ate_badroulbadour.0006 — The Light of the Moon
ate_badroulbadour.0006 = {
	type = character_event
	title = ate_badroulbadour.0006.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { has_sexuality = homosexual }
				desc = ate_badroulbadour.0006.desc.gay
			}
			triggered_desc = {
				trigger = { has_sexuality = asexual }
				desc = ate_badroulbadour.0006.desc.asexual
			}
			desc = ate_badroulbadour.0006.desc
		}
	}
	theme = love

	trigger = {
		exists = var:badroulbadour
		var:badroulbadour = { is_alive = yes }
	}

	immediate = {
		var:badroulbadour = { save_scope_as = badroulbadour_scope }
	}

	left_portrait = {
		character = root
		animation = personality_compassionate
	}
	right_portrait = {
		character = scope:badroulbadour_scope
		animation = flirtatious
		triggered_animation = {
			trigger = { OR = { has_sexuality = homosexual has_sexuality = asexual } }
			animation = personality_content
		}
	}

	option = {
		name = ate_badroulbadour.0006.a
		add_character_flag = badroulbadour_done
		remove_variable = badroulbadour
		add_prestige = 100
		dynasty = { add_prestige = 100 }
		scope:badroulbadour_scope = { add_trait = loyal }
		scope:badroulbadour_scope = { set_house = house:house_ElShabaz }
		if = {
			limit = {
				OR = { has_sexuality = homosexual has_sexuality = asexual }
				any_relation = { type = best_friend }
			}
			set_relation_friend = scope:badroulbadour_scope
		}
		else_if = {
			limit = { OR = { has_sexuality = homosexual has_sexuality = asexual } }
			set_relation_best_friend = scope:badroulbadour_scope
		}
		else_if = {
			limit = { any_relation = { type = soulmate } }
			set_relation_lover = scope:badroulbadour_scope
		}
		else = {
			set_relation_soulmate = scope:badroulbadour_scope
		}
		stress_impact = {
			compassionate = miniscule_stress_impact_loss
		}
	}
}
