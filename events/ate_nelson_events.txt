# ate_nelson.0010 — Make Way for Prince Ali
ate_nelson.0010 = {
	type = character_event
	title = ate_nelson.0010.t
	desc = ate_nelson.0010.desc
	theme = party

	immediate = {
		create_character = {
			name = "Will"
			gender = male
			faith = faith:shriner
			culture = culture:sunshiner
			age = 42
			employer = root
			trait = education_stewardship_2
			trait = gregarious
			trait = zealous
			trait = eccentric
			trait = shrewd
			trait = poet
			trait = lifestyle_mystic
			save_scope_as = will_williams_scope
		}
		set_variable = { name = will_williams value = scope:will_williams_scope }
		scope:will_williams_scope = { give_nickname = nick_the_genius }
		scope:will_williams_scope = { set_house = house:house_Williams }
		random_courtier = {
			limit = { NOT = { this = scope:will_williams_scope } }
			save_scope_as = crowd_1
		}
		random_courtier = {
			limit = {
				NOT = { this = scope:crowd_1 }
				NOT = { this = scope:will_williams_scope }
			}
			save_scope_as = crowd_2
		}
	}

	left_portrait = {
		character = root
		animation = personality_content
	}
	right_portrait = {
		character = scope:will_williams_scope
		animation = happiness
	}
	lower_left_portrait = {
		character = scope:crowd_1
		animation = happiness
	}
	lower_right_portrait = {
		character = scope:crowd_2
		animation = happiness
	}

	option = {
		name = ate_nelson.0010.a
		trigger_event = { id = ate_nelson.0011 days = 1 }
