	theme = stewardship

		character = root
		animation = personality_compassionate

	# Accept — embrace the Neomoor identity
	option = {
		name = ate_nelson.0001.a
		add_character_flag = nelson_neomoor_done
		set_culture = culture:neomoor

	}

	# Decline — not yet
	option = {
		name = ate_nelson.0001.b
	}
}

# ——— Badroulbadour Chain ———
# Triggered by nelson_badroulbadour_chain_check on_yearly_pulse

# ate_nelson.0003 — A Captivating Dance (first encounter)
ate_nelson.0003 = {
	type = character_event
	title = ate_nelson.0003.t
	desc = ate_nelson.0003.desc
	theme = intrigue

	immediate = {
		create_character = {
			name = "Badroulbadour"
			gender = female
			faith = faith:shriner
			culture = culture:neomoor
			age = 22
			trait = beauty_good_3
			trait = intellect_good_2
			trait = gregarious
			employer = root
			save_scope_as = badroulbadour_scope
		}
		set_variable = { name = badroulbadour value = scope:badroulbadour_scope }
		add_character_flag = nelson_badroulbadour_started
	}

	left_portrait = {
		character = root
		animation = idle
	}
	right_portrait = {
		character = scope:badroulbadour_scope
		animation = personality_content
	}

	# Speak with her
	option = {
		name = ate_nelson.0003.a
		trigger_event = { id = ate_nelson.0004 days = 1 }
	}

	# Walk away
	option = {
		name = ate_nelson.0003.b
		scope:badroulbadour_scope = { move_to_pool = yes }
		remove_variable = badroulbadour
		remove_character_flag = nelson_badroulbadour_started
	}
}
