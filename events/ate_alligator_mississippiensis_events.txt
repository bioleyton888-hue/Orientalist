namespace = ate_floridian_alligator

# Floridian Pet alligator chain
#	ate_floridian_alligator.0001 - A Swamp Hatchling (discovery)
#	ate_floridian_alligator.0002 - Name the alligator
#	ate_floridian_alligator.0003 - The alligator settles in (modifier granted)
#
# Council follow-up chain
#	ate_floridian_alligator.0004 - Bring the gator to the council meeting?
#	ate_floridian_alligator.0005 - (A 75%) Gator intimidates the least-loyal councillor
#	ate_floridian_alligator.0006 - (A 25%) Gator bites the least-loyal councillor
#	ate_floridian_alligator.0007 - (B 50%) Gator plays with a courtier's children
#	ate_floridian_alligator.0008 - (B 50%) Gator eats a servant
#
# Death event
#	ate_floridian_alligator.0016 - The Old Gator (fires after 40 years; driven by the story cycle death effect_group)
#
# Flavor chain (driven by the ate_story_cycle_pet_alligator flavor effect_group)
#	ate_floridian_alligator.0009 - A Foreign Guest and the Gator (diplomacy duel)
#	ate_floridian_alligator.0010 - The Gator Escapes (prowess duel)
#	ate_floridian_alligator.0011 - The Gator Won't Eat
#	ate_floridian_alligator.0012 - Someone Wants to Buy It (intrigue duel; sell ends the story)
#	ate_floridian_alligator.0013 - The Gator and the Prisoner

# Cold-Blooded Defiant — thrown from your horse into gator territory; one rolls belly-up
ate_floridian_alligator.0001 = {
	type = character_event
	title = ate_floridian_alligator.0001.t
	desc = ate_floridian_alligator.0001.desc
	theme = pet
	left_portrait = {
		character = root
		animation = shock
	}
	override_background = { reference = forest }

	trigger = {
		is_ai = no
		culture = { has_cultural_pillar = heritage_floridian }
		NOT = { has_character_modifier = ate_pet_alligator_modifier }
		NOT = { has_character_flag = ate_gator_declined }
		is_available_healthy_adult = yes
	}

	immediate = {
		play_music_cue = "mx_cue_low_key_positive"
	}

	option = {
		name = ate_floridian_alligator.0001.a
		create_story = ate_story_cycle_pet_alligator
		stress_impact = {
			compassionate = minor_stress_impact_loss
			eccentric = minor_stress_impact_loss
			gregarious = minor_stress_impact_loss
			ambitious = minor_stress_impact_loss
			craven = minor_stress_impact_gain
		}
		ai_chance = { base = 100 }
	}

	option = {
		name = ate_floridian_alligator.0001.b
		add_character_flag = {
			flag = ate_gator_declined
			years = 10
		}
		stress_impact = {
			craven = minor_stress_impact_loss
			brave = minor_stress_impact_gain
		}
		ai_chance = { base = 0 }
	}

	option = {
		name = ate_floridian_alligator.0001.c
		add_dread = 5
		add_gold = minor_gold_value
		add_character_flag = {
			flag = ate_gator_declined
			years = 10
		}
		stress_impact = {
			sadistic = minor_stress_impact_loss
			callous = minor_stress_impact_loss
			brave = minor_stress_impact_loss
			compassionate = minor_stress_impact_gain
		}
		ai_chance = { base = 0 }
	}

	# Secret option — only cannibals see this. A fellow flesh-eater. Keeps the gator.
	option = {
		name = ate_floridian_alligator.0001.d
		trigger = {
			has_trait = cannibal
		}
		create_story = ate_story_cycle_pet_alligator
		stress_impact = {
			cannibal = major_stress_impact_loss
		}
		ai_chance = { base = 0 }
	}
}

# Name your alligator
ate_floridian_alligator.0002 = {
	type = character_event
	title = ate_floridian_alligator.0002.t
	desc = ate_floridian_alligator.0002.desc
	theme = pet
	left_portrait = {
		character = root
		animation = happiness
	}
	override_background = { reference = forest }

	immediate = {
		add_character_flag = ate_gator_default_wally
		add_character_flag = ate_gator_default_ali
		add_character_flag = ate_gator_default_waylon
		add_character_flag = ate_gator_default_snapper

		if = {
			limit = { is_ai = yes }
			random_list = {
				1 = { set_variable = { name = ate_gator_name  value = flag:ate_gator_name_wally } }
				1 = { set_variable = { name = ate_gator_name  value = flag:ate_gator_name_ali } }
				1 = { set_variable = { name = ate_gator_name  value = flag:ate_gator_name_waylon } }
				1 = { set_variable = { name = ate_gator_name  value = flag:ate_gator_name_snapper } }
			}
		}
	}

	widget = {
		gui = "event_window_widget_enter_text"
		container = "dynamic_birth_name"
		controller = {
			type = text
			data = {
				key = ate_gator_name
				default = {
					localization_key = ate_gator_name_wally
					trigger = { has_character_flag = ate_gator_default_wally }
				}
				default = {
					localization_key = ate_gator_name_ali
					trigger = { has_character_flag = ate_gator_default_ali }
				}
				default = {
					localization_key = ate_gator_name_waylon
					trigger = { has_character_flag = ate_gator_default_waylon }
				}
				default = {
					localization_key = ate_gator_name_snapper
					trigger = { has_character_flag = ate_gator_default_snapper }
				}
			}
		}
		setup_scope = {
			root = { save_scope_as = text_target }
		}
	}

	option = {
		name = ate_floridian_alligator.0002.a
		trigger_event = { id = ate_floridian_alligator.0003 }
	}

	after = {
		remove_character_flag = ate_gator_default_wally
		remove_character_flag = ate_gator_default_ali
		remove_character_flag = ate_gator_default_waylon
		remove_character_flag = ate_gator_default_snapper
		if = {
			limit = { NOT = { has_variable = ate_gator_name } }
			set_variable = {
				name = ate_gator_name
				value = flag:ate_gator_name_wally
			}
		}
		remove_character_flag = ate_is_naming_gator
	}
}

# 0009 — A Foreign Guest and the Gator
ate_floridian_alligator.0009 = {
	type = character_event
	title = ate_floridian_alligator.0009.t
	desc = ate_floridian_alligator.0009.desc
	theme = pet
	override_background = { reference = courtyard }
	left_portrait = {
		character = root
		animation = personality_bold
	}
	right_portrait = {
		character = scope:guest
		animation = fear
	}

	trigger = {
		has_character_modifier = ate_pet_alligator_modifier
		is_landed = yes
		any_courtier = {
			is_adult = yes
			NOT = { this = root }
		}
	}

	immediate = {
		random_courtier = {
			limit = {
				is_adult = yes
				NOT = { this = root }
			}
			save_scope_as = guest
		}
	}

	option = {
		name = ate_floridian_alligator.0009.a
		duel = {
			skill = diplomacy
			value = 12
			50 = {
				compare_modifier = { value = scope:duel_value multiplier = 3.5 min = -29 }
				desc = ate_floridian_alligator.0009.a.success.tt
				add_prestige = minor_prestige_gain
				scope:guest = {
					add_opinion = { target = root modifier = respect_opinion opinion = 15 }
				}
				send_interface_toast = {
					type = event_toast_effect_good
					title = ate_floridian_alligator.0009.a.success
					left_icon = root
				}
			}
			50 = {
				compare_modifier = { value = scope:duel_value multiplier = -3.5 min = -19 }
				desc = ate_floridian_alligator.0009.a.fail.tt
				add_stress = minor_stress_gain
				scope:guest = {
					add_opinion = { target = root modifier = scared_opinion opinion = -15 }
				}
				send_interface_toast = {
					type = event_toast_effect_bad
					title = ate_floridian_alligator.0009.a.fail
					left_icon = root
				}
			}
		}
	}

	option = {
		name = ate_floridian_alligator.0009.b
		add_prestige = minor_prestige_loss
		scope:guest = {
			add_opinion = { target = root modifier = friendliness_opinion opinion = 5 }
		}
	}

	# Cruel option — set the gator on the guest.
	option = {
		name = ate_floridian_alligator.0009.c
		trigger = {
			OR = {
				has_trait = sadistic
				has_trait = callous
				has_trait = torturer
			}
		}
		scope:guest = {
			add_character_modifier = {
				modifier = ate_gator_bite_modifier
				years = 2
			}
			add_opinion = {
				target = root
				modifier = hate_opinion
				opinion = -30
			}
		}
		add_dread = 20
	}

	# Cannibal option — feed the guest to the gator, keep the best cuts.
	option = {
		name = ate_floridian_alligator.0009.d
		trigger = {
			has_trait = cannibal
		}
		scope:guest = {

				death_reason = death_eaten

		}

	}
